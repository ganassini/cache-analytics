```bash
 $ git clone https://github.com/ganassini/cache-sim

 $ git clone https://github.com/toddmaustin/simplesim-3.0
 $ cd simplesim-3.0
 $ make config-pisa
 $ make
 $ cd ../experiment-N
 $ nano run.sh # modify $DIR variable to match the path
 $ ./generate_data.py
```
